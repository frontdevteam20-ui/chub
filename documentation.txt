# Analytics Dashboard Documentation

## Project Overview
This analytics dashboard provides real-time Google Analytics data visualization with dynamic date range filtering, country-wise user distribution, traffic acquisition analysis, and comprehensive metrics tracking.

---

## Backend Architecture

### Technology Stack
- **Framework**: Express.js (Node.js)
- **Google Analytics API**: Google Analytics Data API v1beta
- **Authentication**: Service Account with JSON credentials
- **CORS**: Enabled for cross-origin requests
- **Environment**: dotenv for configuration management

### Authentication Setup
```javascript
// Local Development - File-based authentication
const analyticsDataClient = new BetaAnalyticsDataClient({
  keyFilename: process.env.GA_KEY_PATH || './service-account-key.json',
});

// Production - Environment variable authentication
const analyticsDataClient = new BetaAnalyticsDataClient({
  credentials: JSON.parse(process.env.GA_SERVICE_ACCOUNT),
});
```

### Environment Variables
- `GA_PROPERTY_ID`: Google Analytics Property ID (e.g., "495923879")
- `GA_KEY_PATH`: Path to service account key file (local development)
- `GA_SERVICE_ACCOUNT`: JSON string of service account credentials (production)
- `PORT`: Server port (default: 4000)

## Troubleshooting Guide

### Common Issues & Solutions

#### 1. "client_email field missing" Error
**Cause**: GA_SERVICE_ACCOUNT not properly formatted
**Solution**: Ensure complete JSON structure in environment variable

#### 2. Port 4000 already in use
**Cause**: Previous backend instance still running
**Solution**: Kill existing process and restart

#### 3. No data returned
**Cause**: Incorrect date range or GA property ID
**Solution**: Verify GA_PROPERTY_ID and date format

#### 4. CORS errors
**Cause**: Frontend and backend on different origins
**Solution**: Proper CORS configuration in backend

```javascript
// Advanced engagement metrics
const calculateEngagementMetrics = (data) => {
  const totalUsers = data.reduce((sum, item) => sum + item.activeUsers, 0);
  const totalSessions = data.reduce((sum, item) => sum + item.sessions, 0);
  const totalEngagementTime = data.reduce((sum, item) => sum + (item.avgEngagementTimePerUser * item.activeUsers), 0);
  
  return {
    totalUsers,
    totalSessions,
    averageEngagementTime: totalUsers > 0 ? (totalEngagementTime / totalUsers).toFixed(2) : 0,
    sessionsPerUser: totalUsers > 0 ? (totalSessions / totalUsers).toFixed(2) : 0,
    engagementRate: totalSessions > 0 ? ((totalUsers / totalSessions) * 100).toFixed(2) : 0
  };
```

### Traffic Source Analysis
```javascript
// Channel performance analysis
const analyzeTrafficSources = (data) => {
  const channelMetrics = {};
  
  data.forEach(item => {
    if (!channelMetrics[item.channel]) {
      channelMetrics[item.channel] = {
        totalUsers: 0,
        totalSessions: 0,
        totalDuration: 0,
        avgBounceRate: 0,
        count: 0
      };
    }
    
    const channel = channelMetrics[item.channel];
    channel.totalUsers += item.totalUsers;
    channel.totalSessions += item.sessions;
    channel.totalDuration += item.averageSessionDuration;
    channel.avgBounceRate += item.bounceRate;
    channel.count++;
  });
  
  // Calculate averages for each channel
  Object.keys(channelMetrics).forEach(channel => {
    const metrics = channelMetrics[channel];
    metrics.avgSessionDuration = metrics.count > 0 ? (metrics.totalDuration / metrics.count).toFixed(2) : 0;
    metrics.avgBounceRate = metrics.count > 0 ? (metrics.avgBounceRate / metrics.count).toFixed(4) : 0;
    metrics.usersPerSession = metrics.totalSessions > 0 ? (metrics.totalUsers / metrics.totalSessions).toFixed(2) : 0;
  });
  
  return channelMetrics;
};
```

## Frontend State Management Patterns

### Advanced Date Range Management
```javascript
// Comprehensive date range state management
const useDateRange = (initialMonths = 6) => {
  const [dateRange, setDateRange] = useState(() => {
    const endDate = new Date();
    const startDate = new Date();
    startDate.setMonth(startDate.getMonth() - initialMonths);
    return { startDate, endDate };
  });
  
  const [customRangeActive, setCustomRangeActive] = useState(false);
  
  const updateDateRange = useCallback((newRange) => {
    setDateRange(newRange);
    setCustomRangeActive(true);
  }, []);
  
  const resetToDefault = useCallback(() => {
    const endDate = new Date();
    const startDate = new Date();
    startDate.setMonth(startDate.getMonth() - initialMonths);
    setDateRange({ startDate, endDate });
    setCustomRangeActive(false);
  }, [initialMonths]);
  
  return {
    dateRange,
    customRangeActive,
    updateDateRange,
    resetToDefault
  };
};
```

### Data Fetching Hook Pattern
```javascript
// Reusable data fetching hook for analytics
const useAnalyticsData = (endpoint, dependencies = []) => {
  const [data, setData] = useState(null);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState(null);
  
  const fetchData = useCallback(async () => {
    setLoading(true);
    setError(null);
    
    try {
      const response = await fetch(endpoint);
      if (!response.ok) throw new Error(`HTTP ${response.status}`);
      
      const result = await response.json();
      setData(result);
      
    } catch (err) {
      setError(err.message);
      console.error(`Analytics API Error [${endpoint}]:`, err);
    } finally {
      setLoading(false);
    }
  }, [endpoint]);
  
  useEffect(() => {
    fetchData();
  }, [fetchData, ...dependencies]);
  
  return { data, loading, error, refetch: fetchData };
};
```

## Optimization Strategies

### API Request Optimization
```javascript
// Efficient API request batching
class AnalyticsOptimizer {
  constructor() {
    this.cache = new Map();
    this.requestQueue = [];
    this.batchSize = 10;
    this.debounceTime = 300;
  }
  
  // Batch multiple requests
  async batchRequests(requests) {
    const batchedRequests = [];
    for (let i = 0; i < requests.length; i += this.batchSize) {
      batchedRequests.push(requests.slice(i, i + this.batchSize));
    }
    
    const results = [];
    for (const batch of batchedRequests) {
      const batchResult = await Promise.all(batch.map(req => this.executeRequest(req)));
      results.push(...batchResult);
    }
    
    return results;
  }
  
  // Debounce rapid requests
  debounceRequest(request) {
    clearTimeout(this.debounceTimer);
    this.debounceTimer = setTimeout(() => {
      this.executeRequest(request);
    }, this.debounceTime);
  }
}
```

```javascript
  trackRequest(endpoint, startTime, endTime, success) {
    this.requests.push({
      endpoint,
      duration: endTime - startTime,
      success,
      timestamp: new Date().toISOString()
    });
    
    // Keep only last 100 requests
    if (this.requests.length > 100) {
      this.requests = this.requests.slice(-100);
    }
  },
  
  getAverageResponseTime(endpoint) {
    const endpointRequests = this.requests.filter(req => req.endpoint === endpoint);
    if (endpointRequests.length === 0) return 0;
    
    const totalTime = endpointRequests.reduce((sum, req) => sum + req.duration, 0);
    return totalTime / endpointRequests.length;
  },
  
  getSuccessRate(endpoint) {
    const endpointRequests = this.requests.filter(req => req.endpoint === endpoint);
    if (endpointRequests.length === 0) return 100;
    
    const successfulRequests = endpointRequests.filter(req => req.success);
    return (successfulRequests.length / endpointRequests.length) * 100;
  }
};

// Usage in API endpoints
app.use((req, res, next) => {
  const startTime = Date.now();
  
  res.on('finish', () => {
    const endTime = Date.now();
    apiPerformanceTracker.trackRequest(req.path, startTime, endTime, res.statusCode < 400);
  });
  
  next();
});
```

```javascript
// Comprehensive input validation for API endpoints
const validateAnalyticsRequest = (req) => {
  const errors = [];
  
  // Date format validation
  const dateRegex = /^\d{4}-\d{2}-\d{2}$/;
  if (req.query.startDate && !dateRegex.test(req.query.startDate)) {
    errors.push('Invalid startDate format. Use YYYY-MM-DD');
  }
  
  if (req.query.endDate && !dateRegex.test(req.query.endDate)) {
    errors.push('Invalid endDate format. Use YYYY-MM-DD');
  }
  
  // Date range validation
  if (req.query.startDate && req.query.endDate) {
    const start = new Date(req.query.startDate);
    const end = new Date(req.query.endDate);
    const now = new Date();
    
    if (start > end) {
      errors.push('Start date cannot be after end date');
    }
    
    if (end > now) {
      errors.push('End date cannot be in the future');
    }
    
    // Limit date range to 2 years
    const maxDate = new Date();
    maxDate.setFullYear(maxDate.getFullYear() - 2);
    if (start < maxDate) {
      errors.push('Date range cannot exceed 2 years');
    }
  }
  
  // Property ID validation
  if (req.query.propertyId && !/^\d+$/.test(req.query.propertyId)) {
    errors.push('Property ID must be numeric');
  }
  
  return {
    isValid: errors.length === 0,
    errors
  };
};

// Usage in endpoints
app.get('/api/analytics/totals', (req, res, next) => {
  const validation = validateAnalyticsRequest(req);
  
  if (!validation.isValid) {
    return res.status(400).json({
      error: 'Validation failed',
      details: validation.errors
    });
  }
  
  next();
});
```

### Rate Limiting
```javascript
// API rate limiting implementation
const rateLimiter = {
  requests: new Map(),
  windowMs: 60000, // 1 minute
  maxRequests: 100,
  
  isAllowed: (clientId) => {
    const now = Date.now();
    const windowStart = Math.floor(now / this.windowMs) * this.windowMs;
    
    if (!this.requests.has(clientId)) {
      this.requests.set(clientId, []);
    }
    
    const clientRequests = this.requests.get(clientId).filter(
      timestamp => timestamp >= windowStart
    );
    
    return clientRequests.length < this.maxRequests;
  },
  
  recordRequest: (clientId) => {
    const now = Date.now();
    if (!this.requests.has(clientId)) {
      this.requests.set(clientId, []);
    }
    
    this.requests.get(clientId).push(now);
  }
};

// Apply rate limiting
app.use((req, res, next) => {
  const clientId = req.ip || req.headers['x-forwarded-for'];
  
  if (rateLimiter.isAllowed(clientId)) {
    rateLimiter.recordRequest(clientId);
    next();
  } else {
    res.status(429).json({
      error: 'Too many requests',
      message: `Rate limit exceeded. Maximum ${rateLimiter.maxRequests} requests per ${rateLimiter.windowMs / 1000} seconds`
    });
  }
});
```
